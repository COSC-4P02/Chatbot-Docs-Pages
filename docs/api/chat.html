# Chat Api - Websocket

wss://ws.chatbot-ai.gq:8001

WebSocket w/ SSL

# Send to Server

ws.send(JSON.stringify({
  time: new Date().getTime(),
  version: 'brock', // brock / game
  agent: 'user',
  type: 'text',
  extra: '',
  msg: "Hello"
}))

# Parameters

version: brock, game

agent: user, bot

type: text, button

# Text Response

{
  'type': 'text',
  'text': 'Hey there!',
  'disableInput': false
}

# Button Response

{
  'type': 'button',
  'text': 'Here are some links',
  'options': [
    {
      'text': 'How to use Google',
      'value': 'https://google.com',
      'action': 'url'
    },
    {
      'text': "Not what I'm looking for",
      'value': 'result_not_match',
      'action': 'postback'
    }
  ],
  'disableInput': true
},

# Parameters

type: text, button, news

disableInput: true, false

options: []

news: []
